$(document).ready(function(){
	var link = $('.menu-mobile-link');
	var link_active = $('.menu-mobile-link_active');
	var menu = $('.menu_min');
	var menu_active = $('.menu_min_active');
	/*var menu_mobile_on = $('.menu-ul');
	var menu_mobile_off = $('.menu-ul-mobile');*/


	link.click(function(){
		/*добавляем класс*/
		link.toggleClass('menu-mobile-link_active');
		/*menu_mobile_on.toggleClass('menu-ul-mobile');*/
		/*menu_mobile_on.removeClass('menu-ul');*/
		menu_min.toggleClass('menu_min_active');

	});	

	link_active.click(function(){
		/*удаляем класс*/
		link.removeClass('menu-mobile-link_active');
		/*menu_mobile_off.toggleClass('menu-ul');*/
		/*menu_mobile_off.removeClass('menu-ul-mobile');*/
		menu_min.removeClass('menu_min_active');
	});

});
